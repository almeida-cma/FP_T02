Tipos de Variáveis

Variáveis Inteiras (inteiro)

Armazenam números inteiros (sem casas decimais)
programa { funcao inicio() { inteiro a, b, soma a = 10 b = 25 soma = a + b escreva("A soma de ", a, " + ", b, " = ", soma) } }

Variáveis Reais (real)

Armazenam números com casas decimais
programa { funcao inicio() { real altura, peso, imc escreva("Digite sua altura: ") leia(altura) escreva("Digite seu peso: ") leia(peso) imc = peso / (altura * altura) escreva("Seu IMC é: ", imc) } }

Variáveis Lógicas (logico)

Armazenam verdadeiro ou falso
programa { funcao inicio() { logico aprovado real nota escreva("Digite a nota: ") leia(nota) aprovado = (nota >= 7) se (aprovado == verdadeiro) { escreva("Aluno APROVADO!") } senao { escreva("Aluno REPROVADO!") } } }

Variáveis Caractere (cadeia)

Armazenam textos e palavras
programa { funcao inicio() { cadeia nome, curso escreva("Digite seu nome: ") leia(nome) escreva("Digite seu curso: ") leia(curso) escreva("Olá, ", nome, "! Bem-vindo ao curso de ", curso) } }

Operadores Aritméticos

+ (soma), - (subtração), * (multiplicação), / (divisão), % (resto)
programa { funcao inicio() { inteiro a, b real resultado escreva("Digite o primeiro número: ") leia(a) escreva("Digite o segundo número: ") leia(b) escreva(a, " + ", b, " = ", a + b, "\n") escreva(a, " - ", b, " = ", a - b, "\n") escreva(a, " * ", b, " = ", a * b, "\n") escreva(a, " / ", b, " = ", a / b) } }
Fluxogramas - Representação Gráfica

Fluxograma: Soma de Dois Valores

Representação gráfica de um algoritmo de soma
programa { funcao inicio() { inteiro a, b, soma escreva("Digite o primeiro valor: ") leia(a) escreva("Digite o segundo valor: ") leia(b) soma = a + b escreva("A soma é: ", soma) } }

Fluxograma: Média com Decisão

Fluxograma com condicional (SE média >= 7 → APROVADO)
programa { funcao inicio() { real n1, n2, media escreva("Nota 1: ") leia(n1) escreva("Nota 2: ") leia(n2) media = (n1 + n2) / 2 se (media >= 7) { escreva("APROVADO! Média: ", media) } senao { escreva("REPROVADO! Média: ", media) } } }
Condicionais - Tomando Decisões

Condicional Simples (SE)

Executa ação apenas se condição for verdadeira
programa { funcao inicio() { inteiro valor escreva("Digite um valor: ") leia(valor) se (valor <= 20) { escreva("O valor digitado é menor ou igual a 20!") } } }

Condicional Composta (SE/SENÃO)

Dois caminhos: se verdadeiro, executa um bloco; senão, executa outro
programa { funcao inicio() { real nota escreva("Digite a nota do aluno: ") leia(nota) se (nota >= 7) { escreva("Parabéns! Aprovado!") } senao { escreva("Reprovado. Estude mais!") } } }

Conectivo Lógico "E" (e)

Todas as condições devem ser verdadeiras
programa { funcao inicio() { logico salario_recebido, nao_chovendo escreva("Recebeu o salário? (verdadeiro/falso): ") leia(salario_recebido) escreva("Não está chovendo? (verdadeiro/falso): ") leia(nao_chovendo) se (salario_recebido == verdadeiro e nao_chovendo == verdadeiro) { escreva("Vou sair para comprar roupas!") } senao { escreva("Vou ficar em casa.") } } }

Conectivo Lógico "OU" (ou)

Pelo menos uma condição deve ser verdadeira
programa { funcao inicio() { logico onibusA, onibusB escreva("O ônibus A passou? (verdadeiro/falso): ") leia(onibusA) escreva("O ônibus B passou? (verdadeiro/falso): ") leia(onibusB) se (onibusA == verdadeiro ou onibusB == verdadeiro) { escreva("Conseguirei chegar no horário!") } senao { escreva("Vou me atrasar!") } } }

Estrutura ESCOLHA (Caso)

Ideal para menus com múltiplas opções
programa { funcao inicio() { inteiro opcao escreva("=== MENU PRINCIPAL ===\n") escreva("1 - Cadastrar produto\n") escreva("2 - Listar produtos\n") escreva("3 - Sair\n") escreva("Escolha uma opção: ") leia(opcao) escolha(opcao) { caso 1: escreva("Você escolheu CADASTRAR") pare caso 2: escreva("Você escolheu LISTAR") pare caso 3: escreva("Saindo do sistema...") pare caso contrario: escreva("Opção inválida!") } } }
Exercícios Resolvidos

Exercício: Desconto Promocional

Aplicar 25% de desconto se código for "VBFMZB" ou "HT2Y8E"
programa { funcao inicio() { cadeia codigo real valor, valor_final escreva("Digite o código promocional: ") leia(codigo) escreva("Digite o valor do produto: R$ ") leia(valor) se (codigo == "VBFMZB" ou codigo == "HT2Y8E") { valor_final = valor - (valor * 0.25) escreva("Código válido! Desconto de 25% aplicado.\n") escreva("Valor com desconto: R$ ", valor_final) } senao { escreva("Código inválido! Sem desconto.\n") escreva("Valor a pagar: R$ ", valor) } } }

Exercício: Média com Nome

Calcular média e exibir nome do aluno
programa { funcao inicio() { cadeia nome real n1, n2, n3, media escreva("Digite o nome do aluno: ") leia(nome) escreva("Nota 1: ") leia(n1) escreva("Nota 2: ") leia(n2) escreva("Nota 3: ") leia(n3) media = (n1 + n2 + n3) / 3 escreva("Aluno: ", nome, "\n") escreva("Média: ", media, "\n") se (media >= 7) { escreva("Situação: APROVADO!") } senao { escreva("Situação: REPROVADO!") } } }
